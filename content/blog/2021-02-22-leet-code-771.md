# Jewels and Stones

~~~java
class Solution {
    public int numJewelsInStones(String jewels, String stones) {
        char[] stoneArray = stones.toCharArray();
        int result = 0;
        for(char j : jewels.toCharArray()){
            for(char s : stoneArray){
                if(s == j){
                    result++;
                }
            }
        }
        return result;
    }
}
~~~
